defined_text = { 
	name = GetScreenCount 
	text = {
		trigger = { 
			check_variable = { THIS.needed_screen_count < THIS.screen_count }
		}
		localization_key = GetScreens_under_def
	}
	text = {
		trigger = { 
			check_variable = { THIS.needed_screen_count = THIS.screen_count }
		}
		localization_key = GetScreens_equal_def
	}
	text = {
		trigger = { 
			check_variable = { THIS.needed_screen_count > THIS.screen_count }
		}
		localization_key = GetScreens_over_def
	}	

}
